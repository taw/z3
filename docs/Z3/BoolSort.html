def from_const(val)
  if val == true
    BoolExpr.new(LowLevel.mk_true, self)
  elsif val == false
    BoolExpr.new(LowLevel.mk_false, self)
  else
    raise Z3::Exception, "Cannot convert #{val.class} to #{self.class}"
  end
end
